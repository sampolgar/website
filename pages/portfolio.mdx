# My Portfolio

import Image from 'next/image';
import styles from './index.module.css';
import profilePhoto from '../public/images/main-color.jpg';
import Link from 'next/link';
import { getPagesUnderRoute } from 'nextra/context';

export const PortfolioIndex = () => {
  return getPagesUnderRoute('/portfolio').map((page, i, arr) => {
    return (
      <div key={page.route} className={styles.container}>
        <div className={styles.heading}>{page.frontMatter?.title}</div>
        <Image
          className={styles.heroImage}
          src={`${page.frontMatter?.ogImage}`}
          alt='picture of author'
          width={1080}
          height={1080}
          priority={true}
        />
        <div className={styles.content}>
          <div className={styles.about}>{page.frontMatter?.description}</div>
          {page.frontMatter?.description2 && (
            <div className={styles.about}>{page.frontMatter?.description2}</div>
          )}
          <Link
            key={page.route}
            className='text-gray-500 dark:text-gray-400'
            href={page.route}
          >
            Read more
          </Link>
        </div>
        {arr.length - 1 !== i && <hr className={styles.hr} />}
      </div>
    );
  });
};

export const App = () => {
  return (
    <div className={styles.body}>
      <PortfolioIndex />
    </div>
  );
};

<App />
